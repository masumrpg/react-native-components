# Radio

A powerful, animated radio button component designed for single-selection scenarios. Built with React Native Reanimated for smooth animations and includes comprehensive group management functionality.

## Features

<CardGrid>
  <Card title="🔘 Single Selection" icon="star">
    Perfect for scenarios where only one option can be selected at a time.
  </Card>
  <Card title="👥 Group Management" icon="setting">
    Built-in radio group functionality for managing single selections across multiple options.
  </Card>
  <Card title="🎨 Multiple Variants" icon="puzzle">
    Support for default, primary, secondary, outline, filled, ghost, and state variants.
  </Card>
  <Card title="✨ Smooth Animations" icon="rocket">

## Basic Usage

```tsx
import { Radio } from 'rnc-theme';
import { useState } from 'react';

function BasicExample() {
  const [selectedValue, setSelectedValue] = useState('');

  return (
    <View style={styles.container}>
      <Radio.Group
        value={selectedValue}
        onValueChange={setSelectedValue}
      >
        <Radio value="option1">
          <Radio.Label>Option 1</Radio.Label>
        </Radio>

        <Radio value="option2">
          <Radio.Label>Option 2</Radio.Label>
        </Radio>

        <Radio value="option3">
          <Radio.Label>Option 3</Radio.Label>
        </Radio>
      </Radio.Group>

      <Text style={styles.result}>
        Selected: {selectedValue || 'None'}
      </Text>
    </View>
  );
}
```

### Multi-Step Selection

```tsx
function MultiStepSelectionExample() {
  const [step, setStep] = useState(1);
  const [selections, setSelections] = useState({
    category: '',
    subcategory: '',
    specific: '',
  });

  const categories = [
    { value: 'electronics', label: 'Electronics' },
    { value: 'clothing', label: 'Clothing' },
    { value: 'books', label: 'Books' },
  ];

  const subcategories = {
    electronics: [
      { value: 'phones', label: 'Smartphones' },
      { value: 'laptops', label: 'Laptops' },
      { value: 'accessories', label: 'Accessories' },
    ],
    clothing: [
      { value: 'mens', label: "Men's Clothing" },
      { value: 'womens', label: "Women's Clothing" },
      { value: 'kids', label: "Kids' Clothing" },
    ],
    books: [
      { value: 'fiction', label: 'Fiction' },
      { value: 'nonfiction', label: 'Non-Fiction' },
      { value: 'textbooks', label: 'Textbooks' },
    ],
  };

  const handleCategoryChange = (category: string) => {
    setSelections(prev => ({
      ...prev,
      category,
      subcategory: '',
      specific: ''
    }));
    setStep(2);
  };

  const handleSubcategoryChange = (subcategory: string) => {
    setSelections(prev => ({ ...prev, subcategory }));
    setStep(3);
  };

  const resetSelection = () => {
    setSelections({ category: '', subcategory: '', specific: '' });
    setStep(1);
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Product Category Selection</Text>

      {/* Progress indicator */}
      <View style={styles.progressContainer}>
        <Text style={styles.progressText}>Step {step} of 3</Text>
        <View style={styles.progressBar}>
          <View style={[styles.progressFill, { width: `${(step / 3) * 100}%` }]} />
        </View>
      </View>

      {/* Step 1: Category */}
      {step === 1 && (
        <View style={styles.stepContainer}>
          <Text style={styles.stepTitle}>Choose a Category:</Text>

          <Radio.Group
            value={selections.category}
            onValueChange={handleCategoryChange}
          >
            {categories.map((category) => (
              <Radio
                key={category.value}
                value={category.value}
                variant="primary"
              >
                <Radio.Label>{category.label}</Radio.Label>
              </Radio>
            ))}
          </Radio.Group>
        </View>
      )}

      {/* Step 2: Subcategory */}
      {step === 2 && selections.category && (
        <View style={styles.stepContainer}>
          <Text style={styles.stepTitle}>
            Choose a {categories.find(c => c.value === selections.category)?.label} subcategory:
          </Text>

          <Radio.Group
            value={selections.subcategory}
            onValueChange={handleSubcategoryChange}
          >
            {subcategories[selections.category as keyof typeof subcategories]?.map((sub) => (
              <Radio
                key={sub.value}
                value={sub.value}
                variant="secondary"
              >
                <Radio.Label>{sub.label}</Radio.Label>
              </Radio>
            ))}
          </Radio.Group>

          <Button
            onPress={() => setStep(1)}
            variant="ghost"
            style={styles.backButton}
          >
            Back to Categories
          </Button>
        </View>
      )}

      {/* Step 3: Summary */}
      {step === 3 && (
        <View style={styles.stepContainer}>
          <Text style={styles.stepTitle}>Selection Summary:</Text>

          <View style={styles.summaryContainer}>
            <Text style={styles.summaryItem}>
              Category: {categories.find(c => c.value === selections.category)?.label}
            </Text>
            <Text style={styles.summaryItem}>
              Subcategory: {subcategories[selections.category as keyof typeof subcategories]?.find(s => s.value === selections.subcategory)?.label}
            </Text>
          </View>

          <View style={styles.actionButtons}>
            <Button
              onPress={() => setStep(2)}
              variant="outline"
              style={styles.actionButton}
            >
              Back
            </Button>

            <Button
              onPress={resetSelection}
              variant="ghost"
              style={styles.actionButton}
            >
              Start Over
            </Button>

            <Button
              onPress={() => console.log('Confirmed:', selections)}
              variant="primary"
              style={styles.actionButton}
            >
              Confirm
            </Button>
          </View>
        </View>
      )}
    </View>
  );
}
```

### Do's and Don'ts

<Tabs>
  <TabItem label="✅ Do's">
    ```tsx
    // ✅ Use for single selection scenarios
    <Radio.Group value={paymentMethod} onValueChange={setPaymentMethod}>
      <Radio value="credit">Credit Card</Radio>
      <Radio value="debit">Debit Card</Radio>
      <Radio value="paypal">PayPal</Radio>
    </Radio.Group>

    // ✅ Provide clear, descriptive labels
    <Radio value="express">
      <Radio.Label>Express Shipping (1-2 days) - $15.99</Radio.Label>
    </Radio>

    // ✅ Use appropriate variants for context
    <Radio value="delete" variant="error">
      <Radio.Label>Permanently delete account</Radio.Label>
    </Radio>

    // ✅ Include accessibility support
    <Radio
      value="premium"
      accessibilityLabel="Premium subscription plan"
      accessibilityHint="Upgrade to premium for additional features"
    >
      <Radio.Label>Premium Plan</Radio.Label>
    </Radio>
    ```
  </TabItem>

  <TabItem label="❌ Don'ts">
    ```tsx
    // ❌ Don't use for multiple selections
    <Radio.Group value={features} onValueChange={setFeatures}>
      <Radio value="feature1">Feature 1</Radio>
      <Radio value="feature2">Feature 2</Radio>
      // Use Checkbox instead for multiple selections
    </Radio.Group>

    // ❌ Don't use vague labels
    <Radio value="option1">
      <Radio.Label>Option 1</Radio.Label>
    </Radio>

    // ❌ Don't mix different sizes inconsistently
    <Radio size="xs">Small option</Radio>
    <Radio size="xl">Large option</Radio>

    // ❌ Don't forget accessibility
    <Radio value="important" /> // No label or accessibility info
    ```
  </TabItem>
</Tabs>

  For large lists of radio options, consider implementing virtualization or pagination to maintain smooth performance.
</Aside>

  The radio component uses optimized spring animations. The animations are automatically disabled on older devices for better performance.
</Aside>

## Troubleshooting

### Common Issues

<Tabs>
  <TabItem label="State Management">
    **Problem:** Radio selection not updating properly

    **Solutions:**
    ```tsx
    // Ensure proper controlled component pattern
    const [selected, setSelected] = useState('');

    <Radio.Group
      value={selected}
      onValueChange={setSelected} // Make sure this is connected
    >
      <Radio value="option1">Option 1</Radio>
      <Radio value="option2">Option 2</Radio>
    </Radio.Group>

    // For individual radios outside groups
    const [checked, setChecked] = useState(false);

    <Radio
      value="standalone"
      checked={checked}
      onCheckedChange={setChecked}
    />
    ```
  </TabItem>

  <TabItem label="Animation Issues">
    **Problem:** Animations not working smoothly

    **Solutions:**
    ```tsx
    // Ensure react-native-reanimated is properly configured
    import 'react-native-reanimated/plugin';

    // Check for conflicting styles
    <Radio
      style={{
        // Avoid transform conflicts
        transform: undefined,
      }}
    />

    // For performance issues on older devices,
    // animations are automatically optimized
    ```
  </TabItem>

  <TabItem label="Styling Issues">
    **Problem:** Custom styles not applying correctly

    **Solutions:**
    ```tsx
    // Use proper style props for different parts
    <Radio
      style={styles.radioContainer} // Container styles
      value="example"
    >
      <Radio.Indicator style={styles.indicator} />
      <Radio.Label style={styles.label}>Label</Radio.Label>
    </Radio>

    // Ensure theme compatibility
    const styles = useThemedStyles((theme) => ({
      radioContainer: {
        backgroundColor: theme.colors.surface,
      },
      label: {
        color: theme.colors.text,
      },
    }));
    ```
  </TabItem>
</Tabs>
