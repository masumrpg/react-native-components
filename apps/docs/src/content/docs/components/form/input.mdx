# Input

A versatile, themeable input component with smooth animations, floating labels, validation states, icons, and specialized input types. Built with React Native Reanimated for fluid user experiences.

## Features

## Basic Usage

```tsx
import { Input } from 'rnc-theme';
import { useState } from 'react';

function BasicExample() {
  const [value, setValue] = useState('');

  return (
    <View style={styles.container}>
      <Input
        label="Email Address"
        placeholder="Enter your email"
        value={value}
        onChange={setValue}
        keyboardType="email-address"
        autoCapitalize="none"
      />
    </View>
  );
}
```

## Specialized Input Types

### Password Input

```tsx
import { Input } from 'rnc-theme';
import { useState } from 'react';

function PasswordExample() {
  const [password, setPassword] = useState('');
  const [confirmPassword, setConfirmPassword] = useState('');

  const passwordsMatch = password === confirmPassword && password.length > 0;

  return (
    <View style={styles.container}>
      <Input
        label="Password"
        placeholder="Enter your password"
        value={password}
        onChange={setPassword}
        isPasswordInput
        helperText="Must be at least 8 characters with numbers and symbols"
      />

      <Input
        label="Confirm Password"
        placeholder="Confirm your password"
        value={confirmPassword}
        onChange={setConfirmPassword}
        isPasswordInput
        state={confirmPassword ? (passwordsMatch ? 'success' : 'error') : 'default'}
        helperText={
          confirmPassword
            ? (passwordsMatch ? 'Passwords match!' : 'Passwords do not match')
            : 'Re-enter your password'
        }
      />
    </View>
  );
}
```

### Search Input

```tsx
import { Input } from 'rnc-theme';
import { useState, useEffect } from 'react';

function SearchExample() {
  const [searchQuery, setSearchQuery] = useState('');
  const [results, setResults] = useState([]);
  const [isSearching, setIsSearching] = useState(false);

  useEffect(() => {
    if (searchQuery.length > 2) {
      setIsSearching(true);
      // Simulate search
      const timer = setTimeout(() => {
        setResults(['Result 1', 'Result 2', 'Result 3']);
        setIsSearching(false);
      }, 500);
      return () => clearTimeout(timer);
    } else {
      setResults([]);
    }
  }, [searchQuery]);

  return (
    <View style={styles.container}>
      <Input
        label="Search"
        placeholder="Search for items..."
        value={searchQuery}
        onChange={setSearchQuery}
        isSearchInput
        helperText={isSearching ? 'Searching...' : `${results.length} results found`}
        rightIcon={isSearching ? <ActivityIndicator size="small" /> : undefined}
      />

      {results.length > 0 && (
        <View style={styles.results}>
          {results.map((result, index) => (
            <Text key={index} style={styles.resultItem}>{result}</Text>
          ))}
        </View>
      )}
    </View>
  );
}
```

### Form Integration

```tsx
import { Input, Button } from 'rnc-theme';
import { useForm } from 'react-hook-form';

function FormIntegrationExample() {
  const { control, handleSubmit, formState: { errors } } = useForm();

  const onSubmit = (data) => {
    console.log('Form submitted:', data);
  };

  return (
    <View style={styles.container}>
      <Controller
        control={control}
        name="email"
        rules={{
          required: 'Email is required',
          pattern: {
            value: /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i,
            message: 'Invalid email address'
          }
        }}
        render={({ field: { onChange, onBlur, value } }) => (
          <Input
            label="Email"
            placeholder="Enter your email"
            value={value}
            onChange={onChange}
            onBlur={onBlur}
            state={errors.email ? 'error' : 'default'}
            error={errors.email?.message}
            keyboardType="email-address"
          />
        )}
      />

      <Controller
        control={control}
        name="password"
        rules={{
          required: 'Password is required',
          minLength: {
            value: 8,
            message: 'Password must be at least 8 characters'
          }
        }}
        render={({ field: { onChange, onBlur, value } }) => (
          <Input
            label="Password"
            placeholder="Enter your password"
            value={value}
            onChange={onChange}
            onBlur={onBlur}
            state={errors.password ? 'error' : 'default'}
            error={errors.password?.message}
            isPasswordInput
          />
        )}
      />

      <Button onPress={handleSubmit(onSubmit)} variant="primary">
        Submit
      </Button>
    </View>
  );
}
```

## API Reference

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Input label text |
| `placeholder` | `string` | - | Placeholder text |
| `variant` | `ComponentVariant` | `'default'` | Visual style variant |
| `size` | `ComponentSize` | `'md'` | Input size |
| `state` | `ComponentState` | `'default'` | Validation state |
| `value` | `string` | - | Controlled input value |
| `defaultValue` | `string` | - | Default input value |
| `onChange` | `(text: string) => void` | - | Change handler |
| `onFocus` | `(event) => void` | - | Focus handler |
| `onBlur` | `(event) => void` | - | Blur handler |
| `helperText` | `string` | - | Helper text below input |
| `error` | `string` | - | Error message |
| `required` | `boolean` | `false` | Mark as required field |
| `disabled` | `boolean` | `false` | Disable input |
| `readOnly` | `boolean` | `false` | Make input read-only |
| `leftIcon` | `React.ReactNode` | - | Left side icon |
| `rightIcon` | `React.ReactNode` | - | Right side icon |
| `onRightIconPress` | `() => void` | - | Right icon press handler |
| `floatingLabel` | `boolean` | `false` | Enable floating label animation |
| `multiline` | `boolean` | `false` | Enable multiline input |
| `numberOfLines` | `number` | `1` | Number of lines for multiline |
| `maxLength` | `number` | - | Maximum character length |
| `showCharacterCount` | `boolean` | `false` | Show character counter |
| `isPasswordInput` | `boolean` | `false` | Enable password input features |
| `isSearchInput` | `boolean` | `false` | Enable search input features |
| `isTextAreaInput` | `boolean` | `false` | Enable textarea input features |
| `animated` | `boolean` | `true` | Enable animations |
| `autoFocus` | `boolean` | `false` | Auto focus on mount |

### Variants

- `default` - Standard input with border
- `outline` - Outlined input with border focus
- `filled` - Filled background input
- `ghost` - Minimal input without background

### States

- `default` - Normal state
- `error` - Error state with red styling
- `success` - Success state with green styling
- `warning` - Warning state with yellow styling
- `info` - Info state with blue styling

### Sizes

- `xs` - Extra small input
- `sm` - Small input
- `md` - Medium input (default)
- `lg` - Large input
- `xl` - Extra large input

### Do's and Don'ts

<Tabs>
  <TabItem label="✅ Do's">

  <TabItem label="❌ Don'ts">
    ```tsx

    // ❌ Wrong keyboard type

    <Input /> // No accessibility labels
    ```
  </TabItem>
</Tabs>

<Aside type="tip" title="Performance Tip">
  Use `animated={false}` for forms with many inputs to improve performance on lower-end devices.
</Aside>

<Aside type="note" title="Floating Labels">
  Floating labels work best with controlled inputs and provide a modern, space-efficient design.
</Aside>

## Troubleshooting

### Common Issues

<Tabs>
  <TabItem label="Animation Issues">
    **Problem:** Animations not working smoothly

    **Solutions:**
    ```tsx
    // Ensure react-native-reanimated is properly configured
    import 'react-native-reanimated/plugin';

    // Disable animations if needed
    <Input animated={false} />

    // Check for conflicting styles
    <Input style={{ transform: [] }} /> // Avoid transform conflicts
    ```
  </TabItem>

  <TabItem label="Validation Issues">
    **Problem:** Validation states not updating

    **Solutions:**
    ```tsx
    // Ensure state is properly controlled
    const [value, setValue] = useState('');
    const [error, setError] = useState('');

    <Input
      value={value}
      onChange={setValue}
      state={error ? 'error' : 'default'}
      error={error}
    />

    // Use useEffect for validation
    useEffect(() => {
      if (value && !isValid(value)) {
        setError('Invalid input');
      } else {
        setError('');
      }
    }, [value]);
    ```
  </TabItem>

  <TabItem label="Focus Issues">
    **Problem:** Input not focusing properly

    **Solutions:**
    ```tsx
    // Use ref for manual focus
    const inputRef = useRef(null);

    <Input
      ref={inputRef}
      onPress={() => inputRef.current?.focus()}
    />

    // Check for disabled state
    <Input disabled={false} />

    // Ensure proper keyboard handling
    <Input
      returnKeyType="next"
      onSubmitEditing={() => nextInputRef.current?.focus()}
    />
    ```
  </TabItem>
</Tabs>
