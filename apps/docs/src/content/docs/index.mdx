## Tentang rnc-theme

## ğŸ’¡ Penggunaan Dasar

```tsx
import React from 'react';
import { View } from 'react-native';
import { RNCProvider, Button, Typography } from 'rnc-theme';

const App = () => {
  return (
    <RNCProvider>
      <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
        <Typography variant="h1">Selamat Datang!</Typography>
        <Button onPress={() => console.log('Tombol ditekan!')}>
          <Typography color="white">Tekan Saya</Typography>
        </Button>
      </View>
    </RNCProvider>
  );
};

export default App;
```
