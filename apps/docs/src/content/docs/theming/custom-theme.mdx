## Creating a Custom Theme

### Basic Custom Theme

```tsx
import { Theme } from 'rnc-theme';

const myCustomTheme: Partial<Theme> = {
  colors: {
    primary: '#6366F1',      // Indigo
    secondary: '#8B5CF6',    // Purple
    background: '#F8FAFC',   // Slate 50
    surface: '#FFFFFF',      // White
    text: '#0F172A',         // Slate 900
    textSecondary: '#64748B', // Slate 500
    border: '#E2E8F0',       // Slate 200
    error: '#EF4444',        // Red 500
    warning: '#F59E0B',      // Amber 500
    success: '#10B981',      // Emerald 500
    info: '#3B82F6',         // Blue 500
    muted: '#94A3B8',        // Slate 400
    accent: '#EC4899',       // Pink 500
    destructive: '#DC2626',  // Red 600
  },
  components: {
    borderRadius: {
      xs: 4,
      sm: 6,
      md: 8,
      lg: 12,
      xl: 16,
      full: 9999,
    },
  },
};
```

## Implementation Methods

### Method 1: Provider Configuration

```tsx
import { RNCProvider } from 'rnc-theme';

const customTheme = {
  light: {
    colors: {
      primary: '#6366F1',
      // ... other light theme colors
    },
  },
  dark: {
    colors: {
      primary: '#818CF8',
      // ... other dark theme colors
    },
  },
};

export default function App() {
  return (
    <RNCProvider customTheme={customTheme}>
      {/* Your app content */}
    </RNCProvider>
  );
}
```

### Method 2: Runtime Theme Updates

```tsx
import { useTheme } from 'rnc-theme';

function ThemeCustomizer() {
  const { updateCustomTheme } = useTheme();

  const applyBrandTheme = () => {
    updateCustomTheme({
      colors: {
        primary: '#6366F1',
        secondary: '#8B5CF6',
        // ... other colors
      },
    });
  };

  return (
    <Button onPress={applyBrandTheme}>
      Apply Brand Theme
    </Button>
  );
}
```

### Method 3: Theme Registry

```tsx
import { themeRegistry } from 'rnc-theme';

// Register custom theme
themeRegistry.register('brand', createBrandTheme);

// Use in component
function ThemeSwitcher() {
  const { setActivePreset } = useTheme();

  return (
    <Button onPress={() => setActivePreset('brand')}>
      Switch to Brand Theme
    </Button>
  );
}
```

## Advanced Customization

### Dynamic Theme Generation

```tsx
import { generateThemeFromColor } from 'rnc-theme';

function ColorBasedTheme() {
  const [primaryColor, setPrimaryColor] = useState('#6366F1');
  const { updateCustomTheme } = useTheme();

  const generateTheme = useCallback((color: string) => {
    const generatedTheme = generateThemeFromColor(color);
    updateCustomTheme(generatedTheme);
  }, [updateCustomTheme]);

  useEffect(() => {
    generateTheme(primaryColor);
  }, [primaryColor, generateTheme]);

  return (
    <ColorPicker
      color={primaryColor}
      onColorChange={setPrimaryColor}
    />
  );
}
```

### Conditional Theme Properties

```tsx
import { Platform } from 'react-native';

const platformAwareTheme = {

```

### Theme Inheritance

```tsx
import { lightTheme } from 'rnc-theme';

// Extend existing theme
const extendedTheme = {
  ...lightTheme,
  colors: {
    ...lightTheme.colors,
    primary: '#6366F1',     // Override primary
    secondary: '#8B5CF6',   // Override secondary
    // Keep all other colors from lightTheme
  },
  components: {
    ...lightTheme.components,
    borderRadius: {
      ...lightTheme.components.borderRadius,
      md: 12, // Override medium border radius
    },
  },
};
```

## Theme Validation

### Built-in Validation

```tsx
import { validateTheme, ThemeValidationError } from 'rnc-theme';

const customTheme = {
  colors: {
    primary: '#invalid-color', // This will fail validation
  },
};

try {
  const isValid = validateTheme(customTheme);
  if (isValid) {
    console.log('Theme is valid!');
  }
} catch (error) {
  if (error instanceof ThemeValidationError) {
    console.error('Theme validation failed:', error.message);
    console.error('Invalid properties:', error.invalidProperties);
  }
}
```

### Custom Validation Rules

```tsx

  const errors: string[] = [];

  // Check brand color requirements

    errors.push('Primary color must be a hex value');
```

### Do's and Don'ts

<Tabs>
  <TabItem label="‚úÖ Do's">

    ```tsx
    // ‚úÖ Use semantic color names
    ```

  <TabItem label="‚ùå Don'ts">
    ```tsx
    // ‚ùå Don't use arbitrary color names
    colors: {
      myFavoriteBlue: '#6366F1',
      thatNiceRed: '#EF4444',
    }

    // ‚ùå Don't use inconsistent spacing
    spacing: {
      xs: 3,
      sm: 7,
      md: 15,
      lg: 23,
    }

    // ‚ùå Don't ignore accessibility
    colors: {
      text: '#CCCCCC',      // Too light
      background: '#FFFFFF', // Poor contrast
    }
    ```
  </TabItem>
